using System;

namespace DevelopmentsOverview
{
   public class Car
    {
        public string Model { get; set; }//марка автомобиля
        public int FuelLevel { get; set; }//сколько бензина в баке

        public delegate void EventDelegate(); //делегат для события

        public event EventDelegate TankIsEmpty;//собитие - бак пустой

        //метод расхода топлива
        public void DriveToShop()
        {
            if (FuelLevel == 0)
            {
                if (TankIsEmpty != null) TankIsEmpty();
            }
            else
                FuelLevel -= 5;
            Console.WriteLine("Съездили в магазин .....Осталось {0} литров бензина\n", FuelLevel);
        }

        public void DriveToWork()
        {
            if (FuelLevel == 0)
            {
                if (TankIsEmpty != null) TankIsEmpty();
            }
            else
                FuelLevel -= 5;
            Console.WriteLine("Съездили на работу .....Осталось {0} литров бензина\n", FuelLevel);
        }

        public  void DriveToHospital()
        {
            
            if (FuelLevel == 0)
            {
                if (TankIsEmpty != null) TankIsEmpty();
            }
            else
                FuelLevel -= 5;
            Console.WriteLine("Съездили в госпиталь ..... Осталось {0} литров бензина\n", FuelLevel);

        }

       
    }
}
