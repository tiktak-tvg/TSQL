using System;

namespace DelegatesDifficult
{
   public static class Services
    {
       //метод-функция калькулятор расхода топлива
       public static double CalculatorConsumptionFuel(double fuelLevel, double fuelConsumption1Km, double distance)
       {
           fuelLevel -= distance * fuelConsumption1Km;
           if (fuelLevel <= 0)
               Console.WriteLine("Машина заглохла на дистанции {0}km - нехватило бензина ", distance);
           else
               Console.WriteLine("Пройдено {0}km - остаток топлива {1} л. ", distance, fuelLevel);
           return fuelLevel;
       }
    }
}

using System;

namespace DelegatesDifficult
{
    class Car
    {
        public string Name { get; set; } //марка машины
        public double FuelLevel { get; set; }//сколько литров топлива
        public double FuelConsumption100Km { get; set; }//расход топлива на 100км
        private double fuelConsumption1Km;//расход топлива на 1км

        //создаем конструктор
        public Car(string name,double fuelLevel,double fuelConsumption100km)
        {
            Name = name;
            FuelLevel = fuelLevel;
            FuelConsumption100Km = fuelConsumption100km;
            fuelConsumption1Km = FuelConsumption100Km/100*1;
        }

        //метод-процедура расчета топлива при движении на дистанцию
        public void Drive(double distance)
        {
            FuelLevel -= distance*fuelConsumption1Km;
            if (FuelLevel <= 0)
                Console.WriteLine("Машина заглохла на дистанции {0}km - нехватило бензина ", distance);          
            else 
                Console.WriteLine("Пройдено {0}km - остаток топлива {1} л. ",distance,FuelLevel);
        }

    }

}
