using System;

namespace InheritanceInclusion
{
    class Office
    {
        public int CabinetNumber { get; set; }

        //включаем класс компьютер в класс офис
        public class Computer
        {
            public int NumberSerial { get; set; }
            //метод имитирующий включение компьютера
            public void TurnOn()
            { 
                //........
            }
        }
        public Computer Comp { get; set; }
        //чтобы он был виден во внешнем коде, его нужно инициализировать как объект в конструкторе Office
        public Office()
        {
            Comp = new Computer();
        }
    }
}
